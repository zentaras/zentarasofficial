generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
}

model User {
  id        String   @id @default(cuid())
  clerkId   String   @unique @map("clerk_id")
  email     String   @unique
  firstName String?  @map("first_name")
  lastName  String?  @map("last_name")
  imageUrl  String?  @map("image_url")
  username  String?  @unique
  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  aiResumeScreenerApps   AiResumeScreenerApplicant[]
  ecommerceAnalyticsApps EcommerceAnalyticsApplicant[]
  sentimentDashboardApps SentimentDashboardApplicant[]
  internshipTracks       InternshipTrack[]

  @@map("users")
}

  priorExperience String    @map("prior_experience") @db.Text
  toolsKnown      String    @map("tools_known")
  availability    String
  startDate       String    @map("start_date")
  status          String    @default("Pending")
  adminNote       String?   @map("admin_note") @db.Text

}

// ── Internship Tracking ───────────────────────────────────────────────────────

model InternshipTrack {
  id          String    @id @default(cuid())
  clerkUserId String    @map("clerk_user_id")
  user        User      @relation(fields: [clerkUserId], references: [clerkId])

  projectKey  String    @map("project_key")   // "ai-resume-screener" | "ecommerce-analytics" | "sentiment-dashboard"
  projectName String    @map("project_name")
  applicantId String    @map("applicant_id")  // FK to the original applicant row

  currentStep Int       @default(1) @map("current_step")  // 1–5
  isCompleted Boolean   @default(false) @map("is_completed")
  completedAt DateTime? @map("completed_at")

  createdAt   DateTime  @default(now()) @map("created_at")
  updatedAt   DateTime  @updatedAt @map("updated_at")

  steps       InternshipStep[]

  @@unique([clerkUserId, applicantId])
  @@index([clerkUserId])
  @@index([projectKey])
  @@map("internship_tracks")
}

model InternshipStep {
  id         String          @id @default(cuid())
  trackId    String          @map("track_id")
  track      InternshipTrack @relation(fields: [trackId], references: [id], onDelete: Cascade)

  stepNumber Int     @map("step_number")  // 1 | 2 | 3 | 4 | 5
  stepTitle  String  @map("step_title")

  // Status: "pending" | "submitted" | "approved" | "rejected"
  status     String  @default("pending")

  // Flexible JSON payload — each step stores its own fields here
  data       Json?

  // Admin review
  adminNote  String?   @map("admin_note") @db.Text
  reviewedAt DateTime? @map("reviewed_at")
  reviewedBy String?   @map("reviewed_by")

  deadline    DateTime? @map("deadline")
  submittedAt DateTime? @map("submitted_at")
  createdAt   DateTime  @default(now()) @map("created_at")
  updatedAt   DateTime  @updatedAt @map("updated_at")

  @@unique([trackId, stepNumber])
  @@index([trackId])
  @@index([status])
  @@map("internship_steps")
}
